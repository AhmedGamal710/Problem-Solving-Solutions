// Loop Solution

function longestWordIn(sentence) {
  let splArray = sentence.split(" ");
  let longestLen = 0;
  let longestWord;

  for (let i = 0; i < splArray.length; i++) {
    if (splArray[i].length > longestLen) {
      longestWord = splArray[i];
      longestLen = splArray[i].length;
    }
  }

  return longestWord;
}
